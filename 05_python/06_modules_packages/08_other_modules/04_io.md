## What is the IO Module?

The `io` module provides Python's main facilities for dealing with various types of I/O (input/output). It includes tools for working with in-memory streams (like files but in RAM), text/binary streams, and the core classes behind Python's `open()` function.

```python
import io

# StringIO - treat string like a file
buffer = io.StringIO()
buffer.write("Hello World")
buffer.seek(0)
content = buffer.read()
print(content)  # Hello World

# BytesIO - treat bytes like a file
byte_buffer = io.BytesIO()
byte_buffer.write(b"Binary data")
byte_buffer.seek(0)
data = byte_buffer.read()
print(data)  # b'Binary data'
```

**Key Characteristics:**
- ✅ In-memory file-like objects
- ✅ Text and binary stream support
- ✅ Base classes for custom I/O
- ✅ Used internally by `open()`
- ✅ Great for testing and data processing

## `StringIO` – In-Memory Text Streams

`StringIO` creates a file-like object that operates on strings in memory instead of on disk.

### Basic Usage

```python
import io

# Create StringIO object
buffer = io.StringIO()

# Write to buffer
buffer.write("Hello ")
buffer.write("World")
buffer.write("\n")
buffer.write("Python")

# Get current position
print(f"Position: {buffer.tell()}")  # Position: 20

# Read from buffer (need to seek to beginning)
buffer.seek(0)
content = buffer.read()
print(content)
# Hello World
# Python

# Get all content without seeking
all_content = buffer.getvalue()
print(repr(all_content))  # 'Hello World\nPython'

# Close the buffer (releases memory)
buffer.close()
```

### Reading Operations

```python
import io

buffer = io.StringIO("Line 1\nLine 2\nLine 3\nLine 4")

# Read entire content
buffer.seek(0)
print(buffer.read())  # All lines

# Read line by line
buffer.seek(0)
print(buffer.readline())  # Line 1
print(buffer.readline())  # Line 2

# Read all lines into list
buffer.seek(0)
lines = buffer.readlines()
print(lines)  # ['Line 1\n', 'Line 2\n', 'Line 3\n', 'Line 4\n']

# Iterate over lines
buffer.seek(0)
for line in buffer:
    print(line.strip())

buffer.close()
```

### StringIO as Context Manager

```python
import io

# Using with statement (auto-closes)
with io.StringIO() as buffer:
    buffer.write("Temporary data")
    buffer.seek(0)
    content = buffer.read()
    print(content)  # Temporary data
# Buffer is automatically closed here
```

## `BytesIO` – In-Memory Binary Streams

`BytesIO` creates a file-like object that operates on bytes in memory.

### Basic Usage

```python
import io

# Create BytesIO object
buffer = io.BytesIO()

# Write bytes
buffer.write(b"Hello ")
buffer.write(b"World")
buffer.write(b"\n")
buffer.write(b"Python")

# Read bytes
buffer.seek(0)
data = buffer.read()
print(data)  # b'Hello World\nPython'

# Get all bytes
all_bytes = buffer.getvalue()
print(all_bytes)  # b'Hello World\nPython'

# Convert to string
text = all_bytes.decode('utf-8')
print(text)  # Hello World\nPython

buffer.close()
```

### Working with Binary Data

```python
import io
import struct

# Write binary data
buffer = io.BytesIO()

# Write integers
buffer.write(struct.pack('i', 42))      # 4 bytes
buffer.write(struct.pack('i', 100))     # 4 bytes
buffer.write(struct.pack('d', 3.14159)) # 8 bytes

# Write raw bytes
buffer.write(b'\x00\x01\x02\x03')

# Read binary data
buffer.seek(0)
num1 = struct.unpack('i', buffer.read(4))[0]
num2 = struct.unpack('i', buffer.read(4))[0]
pi = struct.unpack('d', buffer.read(8))[0]
raw = buffer.read(4)

print(f"num1: {num1}, num2: {num2}, pi: {pi}, raw: {raw.hex()}")

buffer.close()
```

## `TextIOWrapper` – Text Stream Wrapper

`TextIOWrapper` wraps a binary stream with text encoding/decoding.

```python
import io

# Create binary stream
binary_buffer = io.BytesIO()

# Wrap with text encoding
text_wrapper = io.TextIOWrapper(binary_buffer, encoding='utf-8')

# Write text (automatically encoded)
text_wrapper.write("Hello 世界")
text_wrapper.flush()

# Read back
binary_buffer.seek(0)
print(binary_buffer.getvalue())  # b'Hello \xe4\xb8\x96\xe7\x95\x8c'

# Read through wrapper
text_wrapper.seek(0)
content = text_wrapper.read()
print(content)  # Hello 世界

text_wrapper.close()
```

## `open()` – Built-in File Opening

The `open()` function returns a file object that is an instance of `io` classes.

```python
import io

# Text file (returns TextIOWrapper)
f = open('file.txt', 'r')
print(type(f))  # <class '_io.TextIOWrapper'>
f.close()

# Binary file (returns BufferedReader/BufferedWriter)
f = open('file.bin', 'rb')
print(type(f))  # <class '_io.BufferedReader'>
f.close()

# Equivalent using io.open() (same as built-in open)
f = io.open('file.txt', 'r')
print(type(f))  # <class '_io.TextIOWrapper'>
f.close()
```

## Real-World Examples

### Example 1: CSV Generator in Memory

```python
import io
import csv

class CSVGenerator:
    @staticmethod
    def generate_csv(data, headers=None):
        """Generate CSV as string without writing to disk"""
        output = io.StringIO()
        
        writer = csv.writer(output)
        
        if headers:
            writer.writerow(headers)
        
        for row in data:
            writer.writerow(row)
        
        return output.getvalue()
    
    @staticmethod
    def generate_csv_bytes(data, headers=None):
        """Generate CSV as bytes"""
        output = io.BytesIO()
        
        # Wrap BytesIO with text wrapper for CSV writer
        text_wrapper = io.TextIOWrapper(output, encoding='utf-8')
        writer = csv.writer(text_wrapper)
        
        if headers:
            writer.writerow(headers)
        
        for row in data:
            writer.writerow(row)
        
        text_wrapper.flush()
        return output.getvalue()
    
    @staticmethod
    def parse_csv(csv_string):
        """Parse CSV from string"""
        output = io.StringIO(csv_string)
        reader = csv.reader(output)
        return list(reader)

# Usage
data = [
    ['Alice', 30, 'Engineer'],
    ['Bob', 25, 'Designer'],
    ['Charlie', 35, 'Manager']
]

# Generate CSV string
csv_str = CSVGenerator.generate_csv(data, ['Name', 'Age', 'Position'])
print("CSV String:")
print(csv_str)

# Generate CSV bytes
csv_bytes = CSVGenerator.generate_csv_bytes(data, ['Name', 'Age', 'Position'])
print(f"\nCSV Bytes: {csv_bytes[:50]}...")

# Parse CSV back
parsed = CSVGenerator.parse_csv(csv_str)
print(f"\nParsed: {parsed}")
```

### Example 2: JSON Processor in Memory

```python
import io
import json

class JSONProcessor:
    @staticmethod
    def process_large_json(json_string, chunk_size=1024):
        """Process large JSON in chunks using StringIO"""
        buffer = io.StringIO(json_string)
        
        # Read in chunks
        while True:
            chunk = buffer.read(chunk_size)
            if not chunk:
                break
            # Process chunk
            print(f"Read {len(chunk)} bytes")
        
        buffer.close()
    
    @staticmethod
    def stream_json(data):
        """Stream JSON data using StringIO"""
        buffer = io.StringIO()
        
        buffer.write('[')
        for i, item in enumerate(data):
            if i > 0:
                buffer.write(',')
            json.dump(item, buffer)
        buffer.write(']')
        
        buffer.seek(0)
        return json.load(buffer)
    
    @staticmethod
    def format_json(data, indent=2):
        """Format JSON as string"""
        buffer = io.StringIO()
        json.dump(data, buffer, indent=indent)
        return buffer.getvalue()

# Usage
data = [
    {"name": "Alice", "age": 30},
    {"name": "Bob", "age": 25},
    {"name": "Charlie", "age": 35}
]

# Stream JSON
result = JSONProcessor.stream_json(data)
print(f"Streamed: {result}")

# Format JSON
formatted = JSONProcessor.format_json(data, indent=2)
print(f"\nFormatted JSON:\n{formatted}")
```

### Example 3: Log Capture for Testing

```python
import io
import sys
import logging

class LogCapture:
    def __init__(self):
        self.buffer = io.StringIO()
        self.handler = logging.StreamHandler(self.buffer)
    
    def __enter__(self):
        logging.getLogger().addHandler(self.handler)
        return self
    
    def __exit__(self, *args):
        logging.getLogger().removeHandler(self.handler)
        self.buffer.close()
    
    def get_logs(self):
        self.buffer.seek(0)
        return self.buffer.read()
    
    def clear(self):
        self.buffer.seek(0)
        self.buffer.truncate(0)

class StdoutCapture:
    def __init__(self):
        self.buffer = io.StringIO()
        self.original_stdout = sys.stdout
    
    def __enter__(self):
        sys.stdout = self.buffer
        return self
    
    def __exit__(self, *args):
        sys.stdout = self.original_stdout
    
    def get_output(self):
        self.buffer.seek(0)
        return self.buffer.read()

# Usage
print("=== Log Capture ===")
with LogCapture() as capture:
    logging.warning("This is a warning")
    logging.error("This is an error")
    logs = capture.get_logs()
    print(f"Captured logs: {logs}")

print("\n=== Stdout Capture ===")
with StdoutCapture() as capture:
    print("Hello World")
    print("This is captured")
    output = capture.get_output()
    print(f"Captured output: {repr(output)}")
```

### Example 4: File-like Interface for API Responses

```python
import io
import requests
import csv

class APIResponseReader:
    def __init__(self, url):
        self.url = url
        self.buffer = None
    
    def fetch(self):
        """Fetch API response and wrap in StringIO"""
        response = requests.get(self.url)
        self.buffer = io.StringIO(response.text)
        return self
    
    def read_csv(self):
        """Read as CSV"""
        if not self.buffer:
            raise ValueError("No data fetched")
        self.buffer.seek(0)
        return csv.reader(self.buffer)
    
    def read_json(self):
        """Read as JSON"""
        if not self.buffer:
            raise ValueError("No data fetched")
        self.buffer.seek(0)
        import json
        return json.load(self.buffer)
    
    def read_lines(self):
        """Read line by line"""
        if not self.buffer:
            raise ValueError("No data fetched")
        self.buffer.seek(0)
        for line in self.buffer:
            yield line.strip()
    
    def close(self):
        if self.buffer:
            self.buffer.close()

# Usage (simulated)
```

### Example 5: Template Engine with StringIO

```python
import io
import re

class SimpleTemplate:
    def __init__(self, template_string):
        self.template = template_string
    
    def render(self, context):
        """Render template with context"""
        output = io.StringIO()
        
        # Simple variable substitution {{var}}
        def replacer(match):
            var_name = match.group(1)
            return str(context.get(var_name, ''))
        
        rendered = re.sub(r'{{(\w+)}}', replacer, self.template)
        output.write(rendered)
        
        output.seek(0)
        return output.read()
    
    def render_to_file(self, context, filename):
        """Render template directly to file"""
        rendered = self.render(context)
        with open(filename, 'w') as f:
            f.write(rendered)

class TemplateEngine:
    def __init__(self):
        self.templates = {}
    
    def add_template(self, name, template_string):
        """Add template to engine"""
        self.templates[name] = SimpleTemplate(template_string)
    
    def render(self, name, context):
        """Render template by name"""
        if name not in self.templates:
            raise ValueError(f"Template '{name}' not found")
        return self.templates[name].render(context)
    
    def render_string(self, template_string, context):
        """Render inline template"""
        return SimpleTemplate(template_string).render(context)

# Usage
engine = TemplateEngine()
engine.add_template("greeting", "Hello, {{name}}! Welcome to {{app}}.")

context = {"name": "Alice", "app": "Template Engine"}
result = engine.render("greeting", context)
print(result)  # Hello, Alice! Welcome to Template Engine.

# Inline template
inline_result = engine.render_string("User: {{username}}, Age: {{age}}", 
                                     {"username": "bob", "age": 30})
print(inline_result)  # User: bob, Age: 30
```

### Example 6: Data Stream Processor

```python
import io
import gzip
import json

class StreamProcessor:
    @staticmethod
    def compress_text(text, encoding='utf-8'):
        """Compress text using gzip in memory"""
        buffer = io.BytesIO()
        with gzip.GzipFile(fileobj=buffer, mode='wb') as gz:
            gz.write(text.encode(encoding))
        return buffer.getvalue()
    
    @staticmethod
    def decompress_text(compressed_data, encoding='utf-8'):
        """Decompress gzipped data to text"""
        buffer = io.BytesIO(compressed_data)
        with gzip.GzipFile(fileobj=buffer, mode='rb') as gz:
            return gz.read().decode(encoding)
    
    @staticmethod
    def process_large_data(data_generator, chunk_size=1024):
        """Process large data in chunks"""
        buffer = io.BytesIO()
        
        for chunk in data_generator:
            buffer.write(chunk)
            if buffer.tell() >= chunk_size:
                # Process chunk
                buffer.seek(0)
                data = buffer.read()
                yield data
                buffer.seek(0)
                buffer.truncate(0)
        
        # Process remaining
        if buffer.tell() > 0:
            buffer.seek(0)
            yield buffer.read()
        
        buffer.close()
    
    @staticmethod
    def json_stream(data):
        """Stream JSON data with proper handling"""
        buffer = io.StringIO()
        
        buffer.write('[')
        first = True
        for item in data:
            if not first:
                buffer.write(',')
            first = False
            json.dump(item, buffer)
        buffer.write(']')
        
        buffer.seek(0)
        return buffer

# Usage
# Compress text
original = "Hello World! " * 100
compressed = StreamProcessor.compress_text(original)
print(f"Original size: {len(original.encode())} bytes")
print(f"Compressed size: {len(compressed)} bytes")

# Decompress
decompressed = StreamProcessor.decompress_text(compressed)
print(f"Decompressed matches: {original == decompressed}")

# JSON stream
data = [{"id": i, "value": i * 2} for i in range(10)]
stream = StreamProcessor.json_stream(data)
print(f"JSON stream: {stream.read()[:100]}...")
```
