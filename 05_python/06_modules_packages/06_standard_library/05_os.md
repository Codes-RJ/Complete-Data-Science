## What is the OS Module?

The `os` module provides a portable way to interact with the operating system. It allows you to work with files, directories, processes, and environment variables.

```python
import os

# Get current working directory
cwd = os.getcwd()
print(cwd)  # /home/user/project

# List files in directory
files = os.listdir('.')
print(files)  # ['file1.py', 'file2.py', 'folder']

# Create a new directory
os.mkdir('new_folder')

# Check if file exists
print(os.path.exists('file.txt'))  # True/False
```

**Key Characteristics:**
- ✅ Cross-platform (Windows, Linux, Mac)
- ✅ Provides file and directory operations
- ✅ Access to environment variables
- ✅ Process management utilities
- ✅ Path manipulation functions

### Example 1: Directory Tree Visualizer

```python
import os

class DirectoryTree:
    @staticmethod
    def print_tree(path, prefix="", show_hidden=False):
        """Print directory tree structure"""
        try:
            items = os.listdir(path)
            
            # Filter hidden files if needed
            if not show_hidden:
                items = [item for item in items if not item.startswith('.')]
            
            # Sort items (directories first, then files)
            dirs = sorted([item for item in items if os.path.isdir(os.path.join(path, item))])
            files = sorted([item for item in items if os.path.isfile(os.path.join(path, item))])
            items = dirs + files
            
            for i, item in enumerate(items):
                item_path = os.path.join(path, item)
                is_last = i == len(items) - 1
                
                # Choose prefix for this item
                current_prefix = "└── " if is_last else "├── "
                print(f"{prefix}{current_prefix}{item}")
                
                # Recursively print subdirectories
                if os.path.isdir(item_path):
                    next_prefix = prefix + ("    " if is_last else "│   ")
                    DirectoryTree.print_tree(item_path, next_prefix, show_hidden)
        
        except PermissionError:
            print(f"{prefix}[Permission Denied]")
    
    @staticmethod
    def get_tree_string(path, show_hidden=False):
        """Return directory tree as string"""
        import io
        import sys
        
        # Capture print output
        old_stdout = sys.stdout
        sys.stdout = io.StringIO()
        
        print(os.path.basename(path) or path)
        DirectoryTree.print_tree(path, "", show_hidden)
        
        output = sys.stdout.getvalue()
        sys.stdout = old_stdout
        
        return output

# Usage
print("DIRECTORY TREE")
print("=" * 40)
tree = DirectoryTree.get_tree_string('.', show_hidden=False)
print(tree)
```

### Example 2: File Organizer

```python
import os
import shutil

class FileOrganizer:
    FILE_CATEGORIES = {
        'Images': ['.jpg', '.jpeg', '.png', '.gif', '.bmp', '.svg'],
        'Documents': ['.pdf', '.doc', '.docx', '.txt', '.rtf', '.md'],
        'Spreadsheets': ['.xls', '.xlsx', '.csv'],
        'Presentations': ['.ppt', '.pptx'],
        'Code': ['.py', '.js', '.html', '.css', '.java', '.cpp', '.c'],
        'Archives': ['.zip', '.tar', '.gz', '.rar', '.7z'],
        'Videos': ['.mp4', '.avi', '.mov', '.wmv', '.flv'],
        'Music': ['.mp3', '.wav', '.flac', '.aac'],
        'Executables': ['.exe', '.msi', '.app', '.dmg'],
        'Others': []
    }
    
    @classmethod
    def get_category(cls, filename):
        """Get category for a file based on extension"""
        ext = os.path.splitext(filename)[1].lower()
        
        for category, extensions in cls.FILE_CATEGORIES.items():
            if ext in extensions:
                return category
        return 'Others'
    
    @classmethod
    def organize(cls, directory, dry_run=True):
        """Organize files by category"""
        if not os.path.exists(directory):
            print(f"Directory not found: {directory}")
            return
        
        os.chdir(directory)
        files = [f for f in os.listdir('.') if os.path.isfile(f)]
        
        organized = {}
        
        for file in files:
            category = cls.get_category(file)
            organized.setdefault(category, []).append(file)
            
            if not dry_run:
                # Create category folder if it doesn't exist
                if not os.path.exists(category):
                    os.mkdir(category)
                
                # Move file to category folder
                shutil.move(file, os.path.join(category, file))
        
        # Print report
        print(f"\n{'DRY RUN' if dry_run else 'ORGANIZATION'} REPORT")
        print("=" * 40)
        
        for category, files in organized.items():
            print(f"\n{category} ({len(files)} files):")
            for file in files:
                print(f"  - {file}")
        
        return organized

# Usage
print("FILE ORGANIZER")
print("=" * 40)
FileOrganizer.organize('.', dry_run=True)
```

### Example 3: Batch File Renamer

```python
import os
import re

class BatchRenamer:
    @staticmethod
    def add_prefix(directory, prefix, dry_run=True):
        """Add prefix to all files in directory"""
        files = [f for f in os.listdir(directory) if os.path.isfile(os.path.join(directory, f))]
        
        renamed = []
        for file in files:
            old_path = os.path.join(directory, file)
            new_name = f"{prefix}{file}"
            new_path = os.path.join(directory, new_name)
            
            if not dry_run:
                os.rename(old_path, new_path)
            
            renamed.append((file, new_name))
        
        return renamed
    
    @staticmethod
    def add_suffix(directory, suffix, dry_run=True):
        """Add suffix before extension to all files"""
        files = [f for f in os.listdir(directory) if os.path.isfile(os.path.join(directory, f))]
        
        renamed = []
        for file in files:
            name, ext = os.path.splitext(file)
            old_path = os.path.join(directory, file)
            new_name = f"{name}{suffix}{ext}"
            new_path = os.path.join(directory, new_name)
            
            if not dry_run:
                os.rename(old_path, new_path)
            
            renamed.append((file, new_name))
        
        return renamed
    
    @staticmethod
    def replace_text(directory, old_text, new_text, dry_run=True):
        """Replace text in filenames"""
        files = [f for f in os.listdir(directory) if os.path.isfile(os.path.join(directory, f))]
        
        renamed = []
        for file in files:
            if old_text in file:
                old_path = os.path.join(directory, file)
                new_name = file.replace(old_text, new_text)
                new_path = os.path.join(directory, new_name)
                
                if not dry_run:
                    os.rename(old_path, new_path)
                
                renamed.append((file, new_name))
        
        return renamed
    
    @staticmethod
    def sequential_numbers(directory, prefix="file_", start=1, dry_run=True):
        """Rename files with sequential numbers"""
        files = sorted([f for f in os.listdir(directory) if os.path.isfile(os.path.join(directory, f))])
        
        renamed = []
        for i, file in enumerate(files, start):
            name, ext = os.path.splitext(file)
            old_path = os.path.join(directory, file)
            new_name = f"{prefix}{i}{ext}"
            new_path = os.path.join(directory, new_name)
            
            if not dry_run:
                os.rename(old_path, new_path)
            
            renamed.append((file, new_name))
        
        return renamed
```

### Example 4: Disk Usage Analyzer

```python
import os

class DiskAnalyzer:
    @staticmethod
    def get_size(path):
        """Get size of file or directory in bytes"""
        if os.path.isfile(path):
            return os.path.getsize(path)
        
        total = 0
        try:
            for item in os.listdir(path):
                item_path = os.path.join(path, item)
                total += DiskAnalyzer.get_size(item_path)
        except PermissionError:
            pass
        
        return total
    
    @staticmethod
    def format_size(size):
        """Format size in human-readable format"""
        for unit in ['B', 'KB', 'MB', 'GB', 'TB']:
            if size < 1024.0:
                return f"{size:.2f} {unit}"
            size /= 1024.0
        return f"{size:.2f} PB"
    
    @staticmethod
    def analyze(directory, show_hidden=False):
        """Analyze disk usage"""
        if not os.path.exists(directory):
            print(f"Directory not found: {directory}")
            return
        
        items = []
        total_size = 0
        
        try:
            for item in os.listdir(directory):
                if not show_hidden and item.startswith('.'):
                    continue
                
                item_path = os.path.join(directory, item)
                size = DiskAnalyzer.get_size(item_path)
                total_size += size
                
                items.append({
                    'name': item,
                    'path': item_path,
                    'size': size,
                    'is_dir': os.path.isdir(item_path)
                })
        except PermissionError:
            print(f"Permission denied: {directory}")
        
        # Sort by size (largest first)
        items.sort(key=lambda x: x['size'], reverse=True)
        
        return {
            'directory': directory,
            'total_size': total_size,
            'items': items[:10]  # Top 10 largest items
        }
    
    @staticmethod
    def print_report(directory, show_hidden=False):
        """Print disk usage report"""
        result = DiskAnalyzer.analyze(directory, show_hidden)
        
        if not result:
            return
        
        print(f"\nDISK USAGE REPORT: {result['directory']}")
        print("=" * 50)
        print(f"Total size: {DiskAnalyzer.format_size(result['total_size'])}")
        print(f"\nLargest items:")
        print("-" * 50)
        
        for item in result['items']:
            type_icon = "📁" if item['is_dir'] else "📄"
            size_str = DiskAnalyzer.format_size(item['size'])
            percentage = (item['size'] / result['total_size']) * 100 if result['total_size'] > 0 else 0
            print(f"{type_icon} {item['name']:<30} {size_str:>10} ({percentage:.1f}%)")

# Usage
print("DISK USAGE ANALYZER")
print("=" * 40)
DiskAnalyzer.print_report('.', show_hidden=False)
```

### Example 5: File Search Utility

```python
import os
import fnmatch

class FileSearch:
    @staticmethod
    def search_by_name(directory, pattern, recursive=True):
        """Search files by name pattern"""
        matches = []
        
        if recursive:
            for root, dirs, files in os.walk(directory):
                for file in files:
                    if fnmatch.fnmatch(file, pattern):
                        matches.append(os.path.join(root, file))
        else:
            for item in os.listdir(directory):
                item_path = os.path.join(directory, item)
                if os.path.isfile(item_path) and fnmatch.fnmatch(item, pattern):
                    matches.append(item_path)
        
        return matches
    
    @staticmethod
    def search_by_size(directory, min_size=None, max_size=None, recursive=True):
        """Search files by size range"""
        matches = []
        
        for root, dirs, files in os.walk(directory):
            for file in files:
                file_path = os.path.join(root, file)
                size = os.path.getsize(file_path)
                
                if min_size is not None and size < min_size:
                    continue
                if max_size is not None and size > max_size:
                    continue
                
                matches.append((file_path, size))
            
            if not recursive:
                break
        
        return sorted(matches, key=lambda x: x[1], reverse=True)
    
    @staticmethod
    def search_by_extension(directory, extensions, recursive=True):
        """Search files by extension"""
        matches = []
        
        for root, dirs, files in os.walk(directory):
            for file in files:
                ext = os.path.splitext(file)[1].lower()
                if ext in extensions:
                    matches.append(os.path.join(root, file))
            
            if not recursive:
                break
        
        return matches
    
    @staticmethod
    def search_by_content(directory, text, recursive=True):
        """Search files containing specific text"""
        matches = []
        
        for root, dirs, files in os.walk(directory):
            for file in files:
                file_path = os.path.join(root, file)
                try:
                    with open(file_path, 'r', encoding='utf-8') as f:
                        if text in f.read():
                            matches.append(file_path)
                except (UnicodeDecodeError, PermissionError):
                    pass
            
            if not recursive:
                break
        
        return matches

# Usage
print("FILE SEARCH UTILITY")
print("=" * 40)

# Search by name
py_files = FileSearch.search_by_name('.', '*.py', recursive=True)
print(f"Python files found: {len(py_files)}")
for f in py_files[:5]:
    print(f"  {f}")

# Search by size (files larger than 1MB)
```

### Example 6: Directory Synchronization

```python
import os
import shutil
import hashlib
from datetime import datetime

class DirectorySync:
    @staticmethod
    def get_file_hash(filepath):
        """Calculate MD5 hash of file"""
        hash_md5 = hashlib.md5()
        with open(filepath, "rb") as f:
            for chunk in iter(lambda: f.read(4096), b""):
                hash_md5.update(chunk)
        return hash_md5.hexdigest()
    
    @staticmethod
    def get_directory_state(directory):
        """Get state of directory (files with hashes and mtimes)"""
        state = {}
        
        for root, dirs, files in os.walk(directory):
            for file in files:
                file_path = os.path.join(root, file)
                rel_path = os.path.relpath(file_path, directory)
                
                state[rel_path] = {
                    'size': os.path.getsize(file_path),
                    'mtime': os.path.getmtime(file_path),
                    'hash': DirectorySync.get_file_hash(file_path)
                }
        
        return state
    
    @staticmethod
    def sync(src, dst, dry_run=True):
        """Sync source directory to destination directory"""
        print(f"\nSyncing {src} -> {dst}")
        print("=" * 50)
        
        if not os.path.exists(src):
            print(f"Source directory not found: {src}")
            return
        
        if not os.path.exists(dst):
            if not dry_run:
                os.makedirs(dst)
            print(f"Created destination directory: {dst}")
        
        src_state = DirectorySync.get_directory_state(src)
        dst_state = DirectorySync.get_directory_state(dst)
        
        # Files to copy (new or modified)
        to_copy = []
        for rel_path, src_info in src_state.items():
            if rel_path not in dst_state:
                to_copy.append(('new', rel_path))
            elif src_info['hash'] != dst_state[rel_path]['hash']:
                to_copy.append(('modified', rel_path))
        
        # Files to delete (exist in dst but not in src)
        to_delete = [rel_path for rel_path in dst_state if rel_path not in src_state]
        
        # Report
        print(f"\nFiles to copy: {len(to_copy)}")
        for action, rel_path in to_copy[:10]:
            print(f"  {action}: {rel_path}")
        if len(to_copy) > 10:
            print(f"  ... and {len(to_copy) - 10} more")
        
        print(f"\nFiles to delete: {len(to_delete)}")
        for rel_path in to_delete[:10]:
            print(f"  delete: {rel_path}")
        if len(to_delete) > 10:
            print(f"  ... and {len(to_delete) - 10} more")
        
        if dry_run:
            print("\n[DRY RUN] No changes were made")
            return
        
        # Perform copy
        for action, rel_path in to_copy:
            src_path = os.path.join(src, rel_path)
            dst_path = os.path.join(dst, rel_path)
            
            # Ensure destination directory exists
            os.makedirs(os.path.dirname(dst_path), exist_ok=True)
            shutil.copy2(src_path, dst_path)
        
        # Perform delete
        for rel_path in to_delete:
            dst_path = os.path.join(dst, rel_path)
            os.remove(dst_path)
        
        # Remove empty directories
        for root, dirs, files in os.walk(dst, topdown=False):
            if not files and not dirs and root != dst:
                os.rmdir(root)
        
        print(f"\nSync complete: {src} -> {dst}")

# Usage
```
